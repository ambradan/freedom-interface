Euryst: Memory Persistence

Il Problema

Euryst è un AI agent WhatsApp che deve mantenere un layer cognitivo persistente per founder. Il problema centrale è: come memorizzare e recuperare conversazioni, contesto e insights in modo efficiente su WhatsApp?

⚠️ Sfide specifiche:
  • WhatsApp Business API è stateless per design
  • Ogni messaggio è un evento isolato senza contesto automatico
  • Serve memoria a lungo termine (conversazioni passate)
  • Serve memoria a breve termine (contesto conversazione corrente)
  • Serve memoria semantica (pattern, insights, preferenze utente)

2. Edge Function - Message Handler

// supabase/functions/whatsapp-webhook/index.ts

import { createClient } from '@supabase/supabase-js'
import Anthropic from '@anthropic-ai/sdk'

const supabase = createClient(...)
const anthropic = new Anthropic(...)

Deno.serve(async (req) => {
  const { message, from } = await req.json()
  
  // 1. Store incoming message
  await supabase.from('wa_messages').insert({
    user_phone: from,
    direction: 'inbound',
    content: message,
    embedding: await generateEmbedding(message)
  })
  
  // 2. Retrieve context (last 10 messages + semantic search)
  const recentMessages = await getRecentMessages(from, 10)
  const relevantMemories = await semanticSearch(message, from, 5)
  const userProfile = await getUserProfile(from)
  
  // 3. Build prompt with memory
  const systemPrompt = buildSystemPrompt(userProfile, relevantMemories)
  const conversationHistory = formatMessages(recentMessages)
  
  // 4. Generate response
  const response = await anthropic.messages.create({
    model: 'claude-3-5-sonnet-20241022',
    system: systemPrompt,
    messages: conversationHistory,
    max_tokens: 1024
  })
  
  // 5. Store response + update profile
  await storeResponse(from, response.content[0].text)
  await updateUserProfile(from, message, response)
  
  // 6. Send to WhatsApp
  await sendWhatsAppMessage(from, response.content[0].text)
  
  return new Response('ok')
})

async function semanticSearch(query: string, userPhone: string, limit: number) {
  const embedding = await generateEmbedding(query)
  
  const { data } = await supabase.rpc('match_messages', {
    query_embedding: embedding,
    match_threshold: 0.78,
    match_count: limit,
    user_phone: userPhone
  })
  
  return data
}

3. Semantic Search Function (PostgreSQL)

CREATE OR REPLACE FUNCTION match_messages(
  query_embedding VECTOR(1536),
  match_threshold FLOAT,
  match_count INT,
  user_phone TEXT
)
RETURNS TABLE (
  id UUID,
  content TEXT,
  timestamp TIMESTAMPTZ,
  similarity FLOAT
)
LANGUAGE SQL STABLE
AS $$
  SELECT
    id,
    content,
    timestamp,
    1 - (embedding <=> query_embedding) AS similarity
  FROM wa_messages
  WHERE user_phone = match_messages.user_phone
    AND 1 - (embedding <=> query_embedding) > match_threshold
  ORDER BY similarity DESC
  LIMIT match_count;
$$;

Ottimizzazioni

1. Embedding Caching

// Genera embeddings in batch async
// Usa Voyage AI (più economico di OpenAI per embeddings)
const BATCH_SIZE = 100

async function processEmbeddingQueue() {
  const unprocessed = await supabase
    .from('wa_messages')
    .select('id, content')
    .is('embedding', null)
    .limit(BATCH_SIZE)
  
  const embeddings = await voyageai.embed({
    texts: unprocessed.map(m => m.content),
    model: 'voyage-2'
  })
  
  // Update in batch
  await Promise.all(
    unprocessed.map((msg, i) => 
      supabase
        .from('wa_messages')
        .update({ embedding: embeddings[i] })
        .eq('id', msg.id)
    )
  )
}

2. Context Window Management

// Claude 3.5 Sonnet: 200k tokens context
// Strategia: riempi il context con memoria rilevante

function buildContextWindow(userPhone, currentMessage) {
  const budget = 180000 // tokens (lascia margine)
  let used = 0
  
  const context = []
  
  // 1. System prompt + profile (~ 2k tokens)
  context.push(systemPrompt)
  used += 2000
  
  // 2. Conversazione recente (~ 10k tokens)
  const recent = getRecent(userPhone, estimateTokens(budget - used))
  context.push(...recent)
  used += estimateTokens(recent)
  
  // 3. Memorie semantiche rilevanti (~ 5k tokens)
  const semantic = semanticSearch(currentMessage, userPhone, 10)
  context.push(...semantic)
  used += estimateTokens(semantic)
  
  // 4. Insights e pattern (~ 3k tokens)
  const insights = getUserInsights(userPhone)
  context.push(insights)
  
  return context
}
