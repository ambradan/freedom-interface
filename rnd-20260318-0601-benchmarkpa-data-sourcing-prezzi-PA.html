BenchmarkPA: data sourcing prezzi PA

Il Problema

BenchmarkPA necessita di dati pricing affidabili per confrontare prezzi B2B2G nelle acquisizioni sottosoglia della Pubblica Amministrazione. Il data sourcing è critico perché:

  • Frammentazione normativa: Ogni PA può avere convenzioni diverse (CONSIP, Centrali di Acquisto regionali, mercati locali)
  • Opacità dei dati: I prezzi sottosoglia (< €40k) spesso non sono pubblicati in formato strutturato
  • Qualità variabile: Dati incompleti, formati eterogenei, aggiornamenti irregolari
  • Compliance GDPR: Necessità di anonimizzare dati sensibili mantenendo utilità statistica
⚠️ Sfida principale: Le PA non hanno obbligo di pubblicazione per acquisti sottosoglia, rendendo il data sourcing particolarmente complesso.

Schema PostgreSQL Ottimizzato

CREATE TABLE prezzi_pa (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cig VARCHAR(10) UNIQUE,
  importo DECIMAL(12,2) NOT NULL,
  categoria_cpv VARCHAR(10), -- Common Procurement Vocabulary
  descrizione TEXT,
  stazione_appaltante_id INTEGER REFERENCES pa_entities(id),
  geo_location GEOGRAPHY(POINT, 4326), -- PostGIS
  data_pubblicazione DATE,
  data_aggiudicazione DATE,
  fonte VARCHAR(50), -- 'mepa', 'anac', 'consip', 'scraping'
  quality_score DECIMAL(3,2), -- 0.00-1.00
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Indici per performance
CREATE INDEX idx_categoria ON prezzi_pa(categoria_cpv);
CREATE INDEX idx_importo ON prezzi_pa(importo);
CREATE INDEX idx_geo ON prezzi_pa USING GIST(geo_location);
CREATE INDEX idx_data ON prezzi_pa(data_aggiudicazione DESC);

-- Materialized view per analytics
CREATE MATERIALIZED VIEW benchmark_stats AS
SELECT 
  categoria_cpv,
  PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY importo) as q1,
  PERCENTILE_CONT(0.50) WITHIN GROUP (ORDER BY importo) as median,
  PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY importo) as q3,
  AVG(importo) as avg_price,
  COUNT(*) as sample_size
FROM prezzi_pa
WHERE quality_score > 0.7
GROUP BY categoria_cpv;

Data Quality Pipeline

// Edge Function per validazione
const validatePriceData = (record) => {
  let score = 1.0
  
  // Penalità per dati mancanti
  if (!record.cig) score -= 0.3
  if (!record.categoria_cpv) score -= 0.2
  if (!record.geo_location) score -= 0.1
  
  // Anomaly detection
  const stats = await getStatsByCPV(record.categoria_cpv)
  const zscore = (record.importo - stats.avg) / stats.stddev
  if (Math.abs(zscore) > 3) score -= 0.4 // Outlier
  
  // Freshness
  const daysSincePublish = daysBetween(record.data_pubblicazione, new Date())
  if (daysSincePublish > 365) score -= 0.2
  
  return Math.max(0, score)
}
